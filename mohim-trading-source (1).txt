import { pgTable, text, serial, timestamp, pgEnum } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod/v4";

export const sectionEnum = pgEnum("section", ["binary", "forex", "strategy", "general"]);
export const videoTypeEnum = pgEnum("video_type", ["youtube", "uploaded"]);

export const videosTable = pgTable("videos", {
  id: serial("id").primaryKey(),
  title: text("title").notNull(),
  description: text("description"),
  section: sectionEnum("section").notNull().default("general"),
  videoType: videoTypeEnum("video_type").notNull().default("youtube"),
  youtubeUrl: text("youtube_url"),
  fileKey: text("file_key"),
  fileUrl: text("file_url"),
  thumbnailUrl: text("thumbnail_url"),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
});

export const insertVideoSchema = createInsertSchema(videosTable).omit({ id: true, createdAt: true });
export type InsertVideo = z.infer<typeof insertVideoSchema>;
export type Video = typeof videosTable.$inferSelect;

import express, { type Express } from "express";
import cors from "cors";
import session from "express-session";
import pinoHttp from "pino-http";
import router from "./routes";
import { logger } from "./lib/logger";

const app: Express = express();

app.use(
  pinoHttp({
    logger,
    serializers: {
      req(req) {
        return {
          id: req.id,
          method: req.method,
          url: req.url?.split("?")[0],
        };
      },
      res(res) {
        return {
          statusCode: res.statusCode,
        };
      },
    },
  }),
);
app.use(cors({ origin: true, credentials: true }));
app.use(express.json({ limit: "10mb" }));
app.use(express.urlencoded({ extended: true }));

app.use("/api/uploads", express.static("/home/runner/workspace/uploads"));

const sessionSecret = process.env["SESSION_SECRET"];
if (!sessionSecret) {
  throw new Error("SESSION_SECRET environment variable is required");
}

app.use(
  session({
    secret: sessionSecret,
    resave: false,
    saveUninitialized: false,
    cookie: {
      secure: process.env["NODE_ENV"] === "production",
      httpOnly: true,
      maxAge: 24 * 60 * 60 * 1000,
    },
  }),
);

app.use("/api", router);

export default app;

import { Router } from "express";
import { db } from "@workspace/db";
import { videosTable } from "@workspace/db";
import { eq } from "drizzle-orm";
import { CreateVideoBody } from "@workspace/api-zod";

const router = Router();

router.get("/videos", async (req, res) => {
  try {
    const section = req.query["section"] as string | undefined;
    let rows;
    if (section) {
      rows = await db
        .select()
        .from(videosTable)
        .where(eq(videosTable.section, section as "binary" | "forex" | "strategy" | "general"))
        .orderBy(videosTable.createdAt);
    } else {
      rows = await db.select().from(videosTable).orderBy(videosTable.createdAt);
    }
    res.json(rows);
  } catch (err) {
    req.log.error({ err }, "Failed to list videos");
    res.status(500).json({ error: "Failed to fetch videos" });
  }
});

router.post("/videos", async (req, res) => {
  const parsed = CreateVideoBody.safeParse(req.body);
  if (!parsed.success) {
    res.status(400).json({ error: "Invalid request body", details: parsed.error.flatten() });
    return;
  }
  try {
    const [row] = await db.insert(videosTable).values(parsed.data).returning();
    res.status(201).json(row);
  } catch (err) {
    req.log.error({ err }, "Failed to create video");
    res.status(500).json({ error: "Failed to create video" });
  }
});

router.get("/videos/:id", async (req, res) => {
  const id = Number(req.params["id"]);
  if (Number.isNaN(id)) {
    res.status(400).json({ error: "Invalid video ID" });
    return;
  }
  try {
    const [row] = await db.select().from(videosTable).where(eq(videosTable.id, id));
    if (!row) {
      res.status(404).json({ error: "Video not found" });
      return;
    }
    res.json(row);
  } catch (err) {
    req.log.error({ err }, "Failed to get video");
    res.status(500).json({ error: "Failed to fetch video" });
  }
});

router.delete("/videos/:id", async (req, res) => {
  const id = Number(req.params["id"]);
  if (Number.isNaN(id)) {
    res.status(400).json({ error: "Invalid video ID" });
    return;
  }
  try {
    await db.delete(videosTable).where(eq(videosTable.id, id));
    res.status(204).end();
  } catch (err) {
    req.log.error({ err }, "Failed to delete video");
    res.status(500).json({ error: "Failed to delete video" });
  }
});

export default router;

async function sendEmail(subject: string, text: string): Promise<void> {
  const smtpUser = process.env["SMTP_USER"];
  const smtpPass = process.env["SMTP_PASS"];
  if (!smtpUser || !smtpPass) {
    throw new Error(
      "Email not configured. Please set SMTP_USER and SMTP_PASS in your environment secrets.",
    );
  }
  const transporter = nodemailer.createTransport({
    host: "smtp.gmail.com",
    port: 587,
    secure: false,
    auth: { user: smtpUser, pass: smtpPass },
  });
  await transporter.sendMail({ from: smtpUser, to: ADMIN_EMAIL, subject, text });
}

import { Router, type Request, type Response } from "express";
import crypto from "node:crypto";
import multer from "multer";
import path from "node:path";
import fs from "node:fs";

const UPLOAD_DIR = "/home/runner/workspace/uploads";
const CHUNK_DIR = "/home/runner/workspace/uploads/chunks";
for (const d of [UPLOAD_DIR, CHUNK_DIR]) {
  if (!fs.existsSync(d)) fs.mkdirSync(d, { recursive: true });
}

const ALLOWED_EXTENSIONS = new Set([
  ".pdf",
  ".jpg", ".jpeg", ".png", ".gif", ".webp",
  ".mp4", ".webm", ".mov", ".m4v", ".avi", ".mkv",
]);

const fileFilter: multer.Options["fileFilter"] = (_req, file, cb) => {
  const mimeOk = /image\/(jpeg|png|gif|webp)|video\/(mp4|webm|quicktime|x-m4v|x-msvideo|x-matroska)|application\/(pdf|octet-stream|x-pdf)/.test(file.mimetype);
  const extOk = ALLOWED_EXTENSIONS.has(path.extname(file.originalname).toLowerCase());
  if (mimeOk || extOk) cb(null, true);
  else cb(new Error("Only images, videos, and PDFs are allowed"));
};

const singleUpload = multer({
  storage: multer.diskStorage({
    destination: (_req, _file, cb) => cb(null, UPLOAD_DIR),
    filename: (_req, file, cb) => {
      const ext = path.extname(file.originalname).toLowerCase();
      cb(null, `${Date.now()}-${crypto.randomUUID().slice(0, 8)}${ext}`);
    },
  }),
  limits: { fileSize: 20 * 1024 * 1024 },
  fileFilter,
});

const chunkUpload = multer({
  storage: multer.diskStorage({
    destination: (req, _file, cb) => {
      const uploadId = (req.body as Record<string, string>)["uploadId"] ?? "tmp";
      const dir = path.join(CHUNK_DIR, uploadId);
      fs.mkdirSync(dir, { recursive: true });
      cb(null, dir);
    },
    filename: (req, _file, cb) => {
      const idx = (req.body as Record<string, string>)["chunkIndex"] ?? "0";
      cb(null, idx.padStart(6, "0"));
    },
  }),
  limits: { fileSize: 6 * 1024 * 1024 },
});

const router = Router();

router.post("/upload/media", (req: Request, res: Response) => {
  singleUpload.single("file")(req, res, (err) => {
    if (err) {
      const status = err instanceof multer.MulterError && err.code === "LIMIT_FILE_SIZE" ? 413 : 400;
      res.status(status).json({ error: err instanceof Error ? err.message : "Upload failed" });
      return;
    }
    if (!req.file) { res.status(400).json({ error: "No file uploaded" }); return; }
    res.json({ url: `/api/uploads/${req.file.filename}`, mediaType: req.file.mimetype, filename: req.file.originalname });
  });
});

router.post("/upload/chunk", (req: Request, res: Response) => {
  chunkUpload.single("chunk")(req, res, async (err) => {
    if (err) {
      res.status(400).json({ error: err instanceof Error ? err.message : "Chunk upload failed" });
      return;
    }

    const body = req.body as Record<string, string>;
    const uploadId = body["uploadId"];
    const chunkIndex = parseInt(body["chunkIndex"] ?? "0", 10);
    const totalChunks = parseInt(body["totalChunks"] ?? "1", 10);
    const originalName = body["filename"] ?? "upload";
    const mime = body["mime"] ?? "application/octet-stream";

    if (!uploadId || isNaN(chunkIndex) || isNaN(totalChunks)) {
      res.status(400).json({ error: "Missing chunk parameters" });
      return;
    }

    if (chunkIndex < totalChunks - 1) {
      res.json({ received: true, chunkIndex });
      return;
    }

    const chunkDir = path.join(CHUNK_DIR, uploadId);
    const ext = path.extname(originalName).toLowerCase() || ".bin";
    const finalName = `${Date.now()}-${crypto.randomUUID().slice(0, 8)}${ext}`;
    const finalPath = path.join(UPLOAD_DIR, finalName);

    try {
      const writeStream = fs.createWriteStream(finalPath);
      for (let i = 0; i < totalChunks; i++) {
        const chunkPath = path.join(chunkDir, String(i).padStart(6, "0"));
        const data = fs.readFileSync(chunkPath);
        writeStream.write(data);
      }
      await new Promise<void>((resolve, reject) => {
        writeStream.on("finish", resolve);
        writeStream.on("error", reject);
        writeStream.end();
      });
      fs.rmSync(chunkDir, { recursive: true, force: true });
      res.json({ url: `/api/uploads/${finalName}`, mediaType: mime, filename: originalName });
    } catch (e) {
      res.status(500).json({ error: "Failed to assemble file" });
    }
  });
});

export default router;

const CHUNK_SIZE = 4 * 1024 * 1024;
const MAX_FILE_SIZE = 8 * 1024 * 1024 * 1024;

export async function uploadFile(
  file: File,
  onProgress?: (pct: number) => void,
): Promise<{ url: string; mediaType: string }> {
  if (file.size > MAX_FILE_SIZE) {
    throw new Error("File too large. Maximum size is 8GB.");
  }

  if (file.size <= CHUNK_SIZE) {
    onProgress?.(10);
    const form = new FormData();
    form.append("file", file);
    const res = await fetch("/api/upload/media", { method: "POST", body: form });
    const text = await res.text();
    let data: { url?: string; mediaType?: string; error?: string } = {};
    try { data = JSON.parse(text) as typeof data; } catch { data = { error: text.slice(0, 200) || "Server error" }; }
    if (!res.ok) throw new Error(data.error ?? "Upload failed");
    onProgress?.(100);
    return { url: data.url!, mediaType: data.mediaType! };
  }

  const uploadId = crypto.randomUUID();
  const totalChunks = Math.ceil(file.size / CHUNK_SIZE);
  let finalResult: { url: string; mediaType: string } | null = null;

  for (let i = 0; i < totalChunks; i++) {
    const start = i * CHUNK_SIZE;
    const chunk = file.slice(start, Math.min(start + CHUNK_SIZE, file.size));

    const form = new FormData();
    form.append("uploadId", uploadId);
    form.append("chunkIndex", String(i));
    form.append("totalChunks", String(totalChunks));
    form.append("filename", file.name);
    form.append("mime", file.type || "application/octet-stream");
    form.append("chunk", chunk, `chunk-${i}`);

    const res = await fetch("/api/upload/chunk", { method: "POST", body: form });
    const text = await res.text();
    let data: { url?: string; mediaType?: string; received?: boolean; error?: string } = {};
    try { data = JSON.parse(text) as typeof data; } catch { data = { error: text.slice(0, 200) || "Server error" }; }
    if (!res.ok) throw new Error(data.error ?? `Failed at chunk ${i + 1}/${totalChunks}`);

    onProgress?.(Math.round(((i + 1) / totalChunks) * 100));

    if (data.url) finalResult = { url: data.url, mediaType: data.mediaType! };
  }

  if (!finalResult) throw new Error("Upload incomplete");
  return finalResult;
}

export async function compressImageIfNeeded(file: File): Promise<File> {
  if (!file.type.startsWith("image/")) return file;
  return new Promise((resolve) => {
    const img = new window.Image();
    const objectUrl = URL.createObjectURL(file);
    img.onload = () => {
      URL.revokeObjectURL(objectUrl);
      const MAX = 1400;
      let { width, height } = img;
      if (width > MAX) { height = Math.round((height * MAX) / width); width = MAX; }
      if (height > MAX) { width = Math.round((width * MAX) / height); height = MAX; }
      const canvas = document.createElement("canvas");
      canvas.width = width; canvas.height = height;
      const ctx = canvas.getContext("2d");
      if (!ctx) { resolve(file); return; }
      ctx.drawImage(img, 0, 0, width, height);
      canvas.toBlob(
        (blob) => resolve(blob ? new File([blob], file.name.replace(/\.[^.]+$/, ".jpg"), { type: "image/jpeg" }) : file),
        "image/jpeg", 0.85,
      );
    };
    img.onerror = () => { URL.revokeObjectURL(objectUrl); resolve(file); };
    img.src = objectUrl;
  });
}

  const { data: messages = [], refetch } = useListMessages();

  const checkAdmin = useCallback(async () => {
    try {
      const res = await fetch("/api/admin/status", { credentials: "include" });
      const data = await res.json() as { isAdmin: boolean };
      setIsAdmin(data.isAdmin);
    } catch { /* ignore */ }
  }, []);

  useEffect(() => { checkAdmin(); }, [checkAdmin]);

  useEffect(() => {
    const interval = setInterval(() => refetch(), 3000);
    return () => clearInterval(interval);
  }, [refetch]);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  const handleSetName = (e: React.FormEvent) => {
    e.preventDefault();
    if (!inputName.trim()) return;
    const n = inputName.trim();
    setName(n);
    setNameSet(true);
    localStorage.setItem("chatName", n);
  };

  const handleSend = (e: React.FormEvent) => {
    e.preventDefault();
    if (!message.trim() || !name) return;
    sendMessage(
      { data: { authorName: name, content: message.trim() } },
      { onSuccess: () => { setMessage(""); refetch(); } },
    );
  };

  const handleFileChange = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const raw = e.target.files?.[0];
    if (!raw || !name) return;
    e.target.value = "";

    const MAX_SIZE = 8 * 1024 * 1024;
    if (raw.size > MAX_SIZE) {
      alert("File too large. Maximum size is 8GB.");
      return;
    }

    setUploading(true);
    setUploadProgress(0);
    try {
      const file = await compressImageIfNeeded(raw);
      const { url, mediaType } = await uploadFile(file, (pct) => setUploadProgress(pct));
      sendMessage(
        { data: { authorName: name, content: null, mediaUrl: url, mediaType } },
        { onSuccess: () => refetch() },
      );
    } catch (err) {
      alert(err instanceof Error ? err.message : "Upload failed. Please check your connection and try again.");
    } finally {
      setUploading(false);
      setUploadProgress(0);
    }

  };

          <input

              className="w-4 h-4 accent-yellow-400"
            />
            <span className="text-sm text-foreground">Pin this strategy</span>
          </label>
          <div className="flex gap-2 pt-2">
            <button
              onClick={onClose}
              className="flex-1 py-2.5 rounded-xl border border-border bg-secondary text-foreground text-sm font-medium hover:border-primary/30 transition-colors"
            >
              Cancel
            </button>
            <button
              onClick={() => onSave(strategy.id, { ...form, tags: form.tags || null })}
              disabled={!form.title || !form.content}
              className="flex-1 py-2.5 rounded-xl gold-gradient text-black text-sm font-bold disabled:opacity-40 hover:opacity-90 transition-opacity"
            >
              Save Changes
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}

      const res = await fetch("/api/admin/status", { credentials: "include" });

      setIsAdmin(data.isAdmin);
    } catch {
      setIsAdmin(false);
    }

  useEffect(() => { checkStatus(); }, [checkStatus]);

      setVideoUploading(false);
      setVideoUploadPct(0);
    }
  };

  const handleDocFileUpload = async (file: File) => {

    setDocUploadPct(0);

      const compressed = await compressImageIfNeeded(file);

    } catch (err) {
      toast({ title: err instanceof Error ? err.message : "Upload failed. Check your connection.", variant: "destructive" });
    } finally {
      setDocUploading(false);
      setDocUploadPct(0);
    }
  };

    } catch { setCodeError("Connection error"); }
    finally { setCodeLoading(false); }
  };

    setCodeError("");
    try {
      const res = await fetch("/api/admin/code-verify", {
        method: "POST",

        credentials: "include",
        body: JSON.stringify({ code: codeOtp }),

        const d = await res.json() as { token: string };
        setCodeToken(d.token);
        setCodeStep("done");
      } else {
        const d = await res.json() as { error?: string };
        setCodeError(d.error ?? "Invalid code");

    } catch { setCodeError("Connection error"); }
    finally { setCodeLoading(false); }

            }`}
          >
            <Icon className="w-3.5 h-3.5" />

            {key === "messages" && messages.length > 0 && (
              <span className="ml-1 bg-black/20 text-current text-xs rounded-full px-1.5 py-0.5 leading-none">
                {messages.length}
              </span>
            )}
          </button>
        ))}
      </div>

      {(tab === "videos" || tab === "documents") && (

          {sections.map((s) => (

                placeholder="Strategy title"
                value={stratForm.title}
                onChange={(e) => setStratForm({ ...stratForm, title: e.target.value })}
                className="px-3 py-2.5 rounded-xl bg-secondary border border-border text-foreground text-sm placeholder:text-muted-foreground focus:outline-none focus:border-primary/50 transition-colors"
              />
              <select
                value={stratForm.section}
                onChange={(e) => setStratForm({ ...stratForm, section: e.target.value as Section })}
                className="px-3 py-2.5 rounded-xl bg-secondary border border-border text-foreground text-sm focus:outline-none focus:border-primary/50 transition-colors"
              >

  const [editContent, setEditContent] = useState("");
  const [saving, setSaving] = useState(false);

  const [confirmSave, setConfirmSave] = useState(false);

  useEffect(() => {
    if (!token) {
      setError("No access token provided. Generate one from the admin panel.");
      setLoading(false);
      return;

    fetch(`/api/admin/code-view/${token}`)
      .then(async (res) => {
        if (!res.ok) {
          const data = await res.json() as { error?: string };
          throw new Error(data.error ?? "Access denied");
        }
        return res.json() as Promise<CodeFile[]>;
      })
      .then((data) => {
        setFiles(data);
        setActiveFile(data[0]?.path ?? "");

        setError(err instanceof Error ? err.message : "Failed to load code");

      .finally(() => setLoading(false));

import { useState } from "react";
import { Play, Trash2, Youtube, Upload, Link2, Share2, Check } from "lucide-react";
import type { Video } from "@workspace/api-client-react";

function getYouTubeId(url: string): string | null {
  const match = url.match(/(?:youtube\.com\/watch\?v=|youtu\.be\/|youtube\.com\/embed\/)([^&\n?#]+)/);
  return match ? (match[1] ?? null) : null;
}

function getYouTubeThumbnail(url: string): string | null {
  const id = getYouTubeId(url);
  return id ? `https://img.youtube.com/vi/${id}/mqdefault.jpg` : null;
}

interface VideoCardProps {
  video: Video;
  isAdmin?: boolean;
  onDelete?: (id: number) => void;
  onPlay?: (video: Video) => void;
}

export default function VideoCard({ video, isAdmin, onDelete, onPlay }: VideoCardProps) {
  const [copied, setCopied] = useState(false);

  const thumbnail =
    video.thumbnailUrl ||
    (video.youtubeUrl ? getYouTubeThumbnail(video.youtubeUrl) : null);

  const shareUrl = video.youtubeUrl || video.fileUrl || window.location.href;

  const handleCopyLink = async (e: React.MouseEvent) => {
    e.stopPropagation();
    try {
      await navigator.clipboard.writeText(shareUrl);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    } catch {
      /* ignore */
    }
  };

  const handleDeviceShare = async (e: React.MouseEvent) => {
    e.stopPropagation();
    if (navigator.share) {
      try {
        await navigator.share({
          title: video.title,
          text: video.description ?? `Watch: ${video.title}`,
          url: shareUrl,
        });
      } catch {
        /* user cancelled */
      }
    } else {
      await handleCopyLink(e);
    }
  };

  return (
    <div className="group bg-card border border-border rounded-xl overflow-hidden transition-all duration-300 card-glow">
      <div
        className="relative aspect-video bg-muted cursor-pointer"
        onClick={() => onPlay?.(video)}
      >
        {thumbnail ? (
          <img
            src={thumbnail}
            alt={video.title}
            className="w-full h-full object-cover"

// FILE: artifacts/mohim-trading/src/components/strategy-card.tsx

import { useState } from "react";
import { Pin, Trash2, Tag, Clock, ChevronDown, ChevronUp, Link2, Check, Pencil } from "lucide-react";
import { formatDistanceToNow } from "date-fns";
import type { Strategy } from "@workspace/api-client-react";

const PREVIEW_LENGTH = 220;

interface StrategyCardProps {
  strategy: Strategy;
  isAdmin?: boolean;
  onDelete?: (id: number) => void;
  onEdit?: (strategy: Strategy) => void;
}

export default function StrategyCard({ strategy, isAdmin, onDelete, onEdit }: StrategyCardProps) {
  const [expanded, setExpanded] = useState(false);
  const [copied, setCopied] = useState(false);

  const tags = strategy.tags ? strategy.tags.split(",").map((t) => t.trim()).filter(Boolean) : [];
  const timeAgo = formatDistanceToNow(new Date(strategy.createdAt), { addSuffix: true });

  const isLong = strategy.content.length > PREVIEW_LENGTH;
  const displayContent = isLong && !expanded
    ? strategy.content.slice(0, PREVIEW_LENGTH) + "…"
    : strategy.content;

  const handleCopyLink = async () => {
    const url = `${window.location.origin}/strategy#s${strategy.id}`;
    try {
      await navigator.clipboard.writeText(url);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    } catch {
      /* ignore */
    }
  };

  return (
    <div
      id={`s${strategy.id}`}
      className={`bg-card border rounded-xl p-5 transition-all duration-300 card-glow ${
        strategy.isPinned ? "border-primary/40 bg-primary/5" : "border-border"
      }`}
    >
      <div className="flex items-start justify-between gap-3 mb-3">
        <h3 className="font-semibold text-foreground leading-tight">{strategy.title}</h3>
        {strategy.isPinned && (
          <div className="flex-shrink-0 w-6 h-6 rounded-full bg-primary/20 flex items-center justify-center">
            <Pin className="w-3.5 h-3.5 text-primary fill-primary" />
          </div>
        )}
      </div>

      <p className="text-muted-foreground text-sm leading-relaxed mb-1 whitespace-pre-line">
        {displayContent}
      </p>
      {isLong && (
        <button
          onClick={() => setExpanded(!expanded)}
          className="flex items-center gap-1 text-xs text-primary hover:text-primary/80 transition-colors mb-3 mt-1"
        >
          {expanded ? (
            <><ChevronUp className="w-3.5 h-3.5" /> Collapse</>
          ) : (
            <><ChevronDown className="w-3.5 h-3.5" /> Read more</>
          )}
        </button>
      )}
      {!isLong && <div className="mb-3" />}

      {tags.length > 0 && (
        <div className="flex flex-wrap gap-1.5 mb-3">
          {tags.map((tag) => (
            <span
              key={tag}
              className="inline-flex items-center gap-1 px-2 py-0.5 rounded-md bg-accent/50 border border-primary/20 text-accent-foreground text-xs"
            >
              <Tag className="w-2.5 h-2.5" />
              {tag}
            </span>
          ))}
        </div>
      )}

      <div className="flex items-center justify-between flex-wrap gap-2">
        <div className="flex items-center gap-1.5 text-muted-foreground text-xs">
          <Clock className="w-3 h-3" />
          {timeAgo}
        </div>
        <div className="flex items-center gap-3">
          <button
            onClick={handleCopyLink}
            className="flex items-center gap-1 text-xs text-muted-foreground hover:text-primary transition-colors"
          >
            {copied ? <Check className="w-3.5 h-3.5 text-green-500" /> : <Link2 className="w-3.5 h-3.5" />}
            {copied ? "Copied!" : "Copy link"}
          </button>
          {isAdmin && onEdit && (
            <button
              onClick={() => onEdit(strategy)}
              className="flex items-center gap-1 text-xs text-primary hover:text-primary/80 transition-colors"
            >
              <Pencil className="w-3.5 h-3.5" />
              Edit
            </button>
          )}
          {isAdmin && onDelete && (
            <button
              onClick={() => onDelete(strategy.id)}
              className="flex items-center gap-1.5 text-xs text-destructive hover:text-destructive/80 transition-colors"
            >
              <Trash2 className="w-3.5 h-3.5" />
              Delete
            </button>
          )}
        </div>
      </div>
    </div>
  );
}
